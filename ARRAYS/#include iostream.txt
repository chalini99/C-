#include <iostream>
using namespace std;
int min(vector<int>& arr){
  int min=arr[0];
  for (int i=1; i<arr.size(); i++){
     if (arr[i]<min){
        min= arr[i];
     }
  }
return min;
}


int even(vector<int>& arr){
  int count=0;
  for (int i=0; i<arr.size(); i++){
      if (arr[i]%2==0){
         count++;
       }
  }
  return count;
}

       }

}


int positivesum(vector<int>& arr){
   int sum = 0;
   for (int i=0; i<arr.size(); i++){
      if (arr[i] >0){
        sum= sum+ arr[i];
      }
      
   }
   return sum;
}
